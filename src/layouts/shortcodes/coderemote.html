{{/* Shortcode: coderemote Descarga un raw remoto en build-time y lo muestra plegado. - url (req): URL raw (Gist, Git, etc.) - lang (opt): lenguaje. Default: text - title (opt): summary; si no se pasa, usa path.Base(url) - linenos (opt): "table" | "inline" | "false". Default: false - engine (opt): "highlight" | "plain". Default: highlight (inline styles) - open (opt): "1" para abrir por defecto - from/to (opt): rango de líneas (1-based, inclusivo) - debug (opt): "1" muestra tamaño leído */}} {{- $url := .Get "url" -}} {{- $lang := .Get "lang" | default "text" -}} {{- $title := .Get "title" | default (path.Base $url) -}} {{- $linenos := .Get "linenos" | default "false" -}} {{- $engine := .Get "engine" | default "highlight" -}} {{- $open := eq (.Get "open") "1" -}} {{- $debug := eq (.Get "debug") "1" -}} {{- with resources.GetRemote $url }} {{- $content := .Content -}}
{{ $title }}
{{- if $debug -}}
debug: {{ len $content }} bytes
{{- end -}} {{- if eq $engine "plain" -}} 
{{ $content | htmlEscape }}
{{- else -}} {{- highlight $content $lang (dict "linenos" $linenos "anchorLineNos" true "noClasses" true) -}} {{- end -}}
{{- else -}}
⚠️ No se pudo cargar: {{ $url }}
{{- end -}}
