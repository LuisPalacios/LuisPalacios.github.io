{{/* Shortcode: codefile Lee un fichero desde assets/ (recomendado) o ruta física y lo muestra plegado. - path (req): ruta relativa a assets/ (ej: "snippets/hello.cpp"). Fallback: ruta física. - lang (opt): lenguaje (cpp, bash, yaml...). Default: text - title (opt): texto del summary. Default: path - linenos (opt): "table" | "inline" | "false". Default: false - engine (opt): "highlight" | "plain". Default: highlight (Chroma con clases CSS) - open (opt): "1" para abrir el
por defecto - from (opt): línea inicial (1-based). Requiere también "to" - to (opt): línea final (inclusive) - download(opt): "1" para mostrar enlace de descarga si viene de assets/ - debug (opt): "1" muestra tamaño leído Uso: {{< codefile path="snippets/hello.cpp" lang="cpp" title="hello.cpp" open="1" linenos="inline" >}} {{< codefile path="snippets/hello.cpp" lang="cpp" from="5" to="18" >}} */}} {{- $path := .Get "path" -}} {{- $lang := .Get "lang" | default "text" -}} {{- $title := .Get "title" | default $path -}} {{- $linenos := .Get "linenos" | default "false" -}} {{- $engine := .Get "engine" | default "highlight" -}} {{- $open := eq (.Get "open") "1" -}} {{- $showDl := eq (.Get "download") "1" -}} {{- $debug := eq (.Get "debug") "1" -}} {{- $res := resources.Get $path -}} {{- $code := "" -}} {{- if $res }}{{- $code = $res.Content -}} {{- else if fileExists $path }}{{- $code = readFile $path -}} {{- else if fileExists (printf "src/%s" $path) }}{{- $code = readFile (printf "src/%s" $path) -}} {{- end -}}
{{ $title }}
{{- if $debug -}}
debug: {{ len $code }} bytes
{{- end -}} {{- if eq $engine "plain" -}} 
{{ $code | htmlEscape }}
{{- else -}} {{- highlight $code $lang (dict "linenos" $linenos "anchorLineNos" true "noClasses" false) -}} {{- end -}} {{- if and $showDl $res -}} {{- $fp := $res | resources.Fingerprint "sha256" -}}
Download
{{- end -}}
